Procedure colorerCase(chaine col)
debut
	// col_ est l'attribut privé pour la couleur d'une case
	si (col_ = "blanc") alors
		col_ <- col
	sinon
		afficher("Cette case est déjà en couleur")
	fin si
fin

Procedure afficheComposante(Case case1, Case2, chaine col)
variables:
	entier limite
debut
	limite <- longueur - 1
	si (limite > case2.X ou limite > case2.Y) alors
		si (existeCheminCases(case1, case2, col)) alors
			afficher(X et Y)
		fin si
	fin si
	// incX() et incY() incrémente respectivement X et Y
	case2.incX()
	afficheComposante(case1, case2, col)
	case2.incY()
	afficheComposante (case1, case2, col)
